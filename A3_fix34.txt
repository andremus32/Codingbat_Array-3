//Array-3 > fix34
public int[] fix34(int[] nums) {
  
  for(int i = 0; i < nums.length-1; i++){
    if(nums[i] == 3 && nums[i+1] != 4){
    int tmp = nums[i+1];
      for(int j = i+1; j < nums.length; j++){
        if(nums[j] == 4){
          nums[j] = tmp;
        }
    nums[i+1] = 4;
      }
    }
  }
  return nums;
}
